\section{Background}

\begin{frame}{What is XSS?}
\textbf{Cross-Site Scripting (XSS)} means that an attacker gets JavaScript executed in another user's browser \citep{lekies2017scriptgadgets}.

\vspace{0.7em}

\begin{exampleblock}{Classic payload}
\texttt{<script>alert(1)</script>}
\end{exampleblock}

\vspace{0.7em}

\textbf{Goal:} the browser executes attacker-controlled code in the context of the vulnerable website.
\end{frame}

\begin{frame}{Motivation}
Modern defenses usually block \textbf{obvious executable code}:

\vspace{0.5em}

\begin{itemize}
\item \texttt{<script>} tags
\item inline event handlers such as \texttt{onerror} or \texttt{onclick}
\item \texttt{javascript:} URLs
\end{itemize}

\vspace{0.7em}

But modern web applications already contain a lot of \textbf{trusted JavaScript} from frameworks and libraries.
\end{frame}

\begin{frame}[plain]
\centering
\vfill

{\Huge Can an attacker reuse trusted website code\
instead of injecting new JavaScript?\par}

\vfill

{\Large\bfseries Answer:\par}

\vspace{0.8em}

{\LARGE Yes, this is the idea behind\ \textbf{Script Gadgets}.\par}

\vfill
\end{frame}

\section{Script Gadgets}

\begin{frame}{Definition: Script Gadget}
\begin{block}{Script Gadget \citep{lekies2017scriptgadgets}}
A Script Gadget is \textbf{trusted JavaScript already present on a website} that can be triggered by attacker-controlled but seemingly harmless HTML.
\end{block}

\vspace{0.7em}

\begin{itemize}
\item The attacker does \textbf{not} inject the JavaScript gadget itself.
\item The gadget already exists in the application or a library.
\item The attacker only provides input that activates an unsafe code path.
\end{itemize}
\end{frame}

\begin{frame}{But What If No Script Is Injected?}
\begin{center}
\footnotesize
\setlength{\parskip}{0.2em}
\begin{tabular}{@{}c@{}}
\textbf{Classic XSS blocked} \\[0.2em]
$\downarrow$ \\[0.2em]
\textbf{Attacker injects harmless-looking HTML} \\[0.2em]
$\downarrow$ \\[0.2em]
\textbf{Trusted website JavaScript reads it} \\[0.2em]
$\downarrow$ \\[0.2em]
\textbf{Code execution} \\
\end{tabular}
\end{center}

\vspace{0.5em}

\begin{block}{Key idea}
Script Gadgets show that stopping injected scripts is not always enough when the site's own JavaScript does the dangerous work.
\end{block}
\end{frame}

\begin{frame}{Tiny Example}
The attacker injects harmless-looking HTML:

\vspace{0.4em}

\begin{exampleblock}{Injected markup}
\texttt{<div data-cmd="alert(1)"></div>}
\end{exampleblock}

\vspace{0.8em}

The website already contains JavaScript like this:

\vspace{0.4em}

\begin{exampleblock}{Existing trusted code}
\texttt{eval(element.dataset.cmd);}
\end{exampleblock}

\vspace{0.8em}

\textbf{Result:} the attacker did not inject a \texttt{<script>} tag, but code still executes.
\end{frame}

\section{Bypassing Mitigations}

\begin{frame}{Bypassing Filters and Sanitizers}
Filters and sanitizers often block patterns such as:

\vspace{0.4em}

\begin{exampleblock}{Blocked}
\texttt{<script>}, \texttt{onerror}, \texttt{eval}, \texttt{document.cookie}
\end{exampleblock}

\vspace{0.8em}

But a gadget payload may use allowed attributes:

\vspace{0.4em}

\begin{exampleblock}{Allowed-looking}
\texttt{data-html}, \texttt{data-bind}, \texttt{class}, \texttt{id}
\end{exampleblock}

\vspace{0.8em}

These attributes become dangerous only when trusted JavaScript interprets them.
\end{frame}

\begin{frame}{Bypassing CSP}
\textbf{Content Security Policy (CSP)} can block injected scripts.

\vspace{0.7em}

But Script Gadgets reuse scripts that are already trusted by the page.

\vspace{0.7em}

\begin{itemize}
\item \texttt{unsafe-eval}: enables gadgets using \texttt{eval} or template execution
\item \texttt{strict-dynamic}: trusted libraries may create new \texttt{<script>} elements
\item Framework parsers may evaluate attacker-controlled expressions
\end{itemize}

\vspace{0.7em}

\begin{alertblock}{Important result}
In the paper, \textbf{13 out of 16} studied frameworks allowed a bypass of \texttt{strict-dynamic}.
\end{alertblock}
\end{frame}

\begin{frame}{Empirical Results}
Lekies et al. manually analyzed 16 JavaScript libraries and frameworks.

\vspace{0.7em}

\begin{itemize}
\item Gadgets were found in \textbf{almost all} analyzed frameworks.
\item React had no usable gadgets in their study.
\item In the Alexa Top 5000 study, \textbf{19.88\%} of domains had verified gadgets.
\end{itemize}

\vspace{0.7em}

\begin{block}{Takeaway}
Script Gadgets are not only theoretical. They appeared in real frameworks and real websites.
\end{block}

\small \cite{lekies2017scriptgadgets}
\end{frame}

\section{Case Study}

\begin{frame}{Example: Bootstrap Tooltip}
The course example \cite{friendsDemo2026} illustrates the idea with Bootstrap tooltips.

\vspace{0.7em}

\begin{enumerate}
\item A user profile contains unsanitized HTML in a tooltip \texttt{title}.
\item Bootstrap reads the tooltip data.
\item \texttt{sanitize: false} disables Bootstrap's internal protection.
\item Bootstrap renders attacker-controlled HTML.
\item The payload runs when the tooltip is triggered.
\end{enumerate}

\vspace{0.7em}

\begin{block}{Why this is a Script Gadget}
Bootstrap is trusted library code, but it processes attacker-controlled input in an unsafe configuration.
\end{block}
\end{frame}

\section{Conclusion}

\begin{frame}{Countermeasures}
\begin{itemize}
\item Fix the original injection flaw.
\item Use context-aware encoding and sanitization before data enters the DOM.
\item Avoid dangerous APIs such as \texttt{eval}, \texttt{innerHTML}, and \texttt{new Function}.
\item Do not disable library protections such as \texttt{sanitize: true}.
\item Use CSP as defense in depth, not as the only defense.
\end{itemize}
\end{frame}

\begin{frame}{Takeaways}
\begin{alertblock}{Main message}
Harmless-looking HTML can become dangerous when trusted JavaScript processes it unsafely.
\end{alertblock}

\vspace{0.7em}

\begin{itemize}
\item Script Gadgets reuse existing website or library code.
\item They can bypass common XSS mitigations.
\item Trusted frameworks can still be abused.
\item Secure coding is still necessary.
\end{itemize}
